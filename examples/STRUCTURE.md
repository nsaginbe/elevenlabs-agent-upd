    company_description = Column(Text, nullable=True) # extra

    session_system_prompt = Column(Text, nullable=True)

        db.close()

async def test_conversation_with_overrides():
    """Simple test for conversation overrides"""
    
    # Create ElevenLabs client
    elevenlabs = ElevenLabs(api_key=api_key)
    
    # Set up overrides
    conversation_override = {
        "agent": {
            "prompt": {
                "prompt": f"The customer's bank account balance is {customer_balance}. They are based in {customer_location}."
            },
            "first_message": f"Hi {customer_name}, how can I help you today?",
            "language": "en"
        },
        "tts": {
            "voice_id": "21m00Tcm4TlvDq8ikWAM"  # Replace with actual voice ID
        }
    }
    
    # Create configuration with overrides
    config = ConversationInitiationData(
        conversation_config_override=conversation_override
    )
    
    # Initialize conversation
    conversation = Conversation(
        elevenlabs,
        agent_id,
        audio_interface=DefaultAudioInterface(),
        requires_auth=bool(api_key),
        config=config,
        # Test callbacks
        callback_agent_response=lambda response: print(f"Agent: {response}"),
        callback_user_transcript=lambda transcript: print(f"User: {transcript}"),
    )
    
    try:
        # Start the session
        conversation_id = conversation.start_session()
        print(f"Conversation started with ID: {conversation_id}")
        print("Overrides applied successfully!")
        
        # Wait for a few seconds to see the first message
        await asyncio.sleep(5)
        
    except Exception as e:
        print(f"Error: {e}")
    finally:
        # End the session
        conversation.end_session()
        print("Conversation ended.")

if __name__ == "__main__":
